## Jekyll文章头
```
javascript:(function(){const template="---\nlayout: default\ntitle: 文章标题\ndate: 2026-04-17\ncategories: [分类]\ntags: [标签]\n---\n\n";const el=document.activeElement;if(el&&(el.tagName==='TEXTAREA'||el.tagName==='INPUT'&&el.type==='text'||el.isContentEditable)){const start=el.selectionStart;const end=el.selectionEnd;const value=el.value||el.innerText;const newValue=value.slice(0,start)+template+value.slice(end);if(el.value!==undefined){el.value=newValue}else{el.innerText=newValue}el.selectionStart=el.selectionEnd=start+template.length;const event=new Event('input',{bubbles:true});el.dispatchEvent(event)}else{alert('请先点击文本输入框！')}})();
```
